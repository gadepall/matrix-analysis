\begin{align}
\vec{A}=\myvec{-2 \\ -1  },  \vec{B}=\myvec{ 4\\ 0 }, 
\vec{C}=\myvec{3 \\ 3  },  \vec{D}=\myvec{-3 \\ 2  }
\end{align}
and
\begin{align}
	\vec{P} &=\vec{B}-\vec{A}=\myvec{6 \\ 1 \\ }
	\\
	\vec{Q}&=\vec{C}-\vec{D}=\myvec{6 \\ 1 \\ }
	\\
	\vec{R}&=\vec{A}-\vec{C}=\myvec{1 \\ -3 \\ }
	\\
	\vec{S}&=\vec{A}-\vec{D}=\myvec{1 \\ -3 \\ }
\end{align}
Since
$\vec{P}=\vec{Q}$
and 
$\vec{R}=\vec{S}$, from  
	  \eqref{eq:two-pgm},
$ABCD$ is a
